function validateSec(n) {
  let ok = true;
  if (n === 1) {
    ok = req('childName') && req('childAge', true) && reqCustom('std', getStd()) && true;
  }
  if (n === 2) {
    ok = req('parentName') && req('parentPhone') && true;
    if (F.email) {
      const emailOk = /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(document.getElementById('parentEmail').value);
      if (!emailOk) { showErr('parentEmail', true); ok = false; }
      else showErr('parentEmail', false);
    }
  }
  if (n === 3) {
    ok = reqCustom('conf', getSelected('confGrid'));
  }
  if (n === 4) {
    if (!F.consent) { showErr('consent', true); ok = false; }
    else showErr('consent', false);
  }
  return ok;
}
function req(id, isSelect) {
  const el = document.getElementById(id);
  const val = el ? el.value.trim() : '';
  const ok = val.length > 0;
  showErr(id, !ok);
  if (!ok) el.classList.add('err'); else el.classList.remove('err');
  return ok;
}

  const ok = val && val !== '';
  showErr(errId, !ok);

  pick(standard, mode, level) {
    const std  = Math.max(1, Math.min(6, standard || 1));
    const pool = this.STANDARD_TOPICS[std];
    const filter = this.MODE_FILTER[mode];
    let eligible = filter
      ? pool.filter(t => filter.includes(t))
      : pool;
    if (!eligible.length) eligible = pool; // fallback

    // Weighted: 60% current level topics, 30% review (easier), 10% preview (harder)
    const roll = Math.random();
    let topicId;
    if (roll < 0.60 || eligible.length === 1) {
      topicId = eligible[randInt(0, eligible.length - 1)];
    } else if (roll < 0.90 && std > 1) {
      const lowerPool = this.STANDARD_TOPICS[std - 1];
      topicId = lowerPool[randInt(0, lowerPool.length - 1)];
    } else if (std < 6) {
      const higherPool = this.STANDARD_TOPICS[std + 1];
      topicId = higherPool[randInt(0, higherPool.length - 1)];
    } else {
      topicId = eligible[randInt(0, eligible.length - 1)];
    }

    const topic = this.TOPICS[topicId];
    if (!topic) return null;
    const q = topic.gen(std);
    q.topicId    = topicId;
    q.topicLabel = topic.label;
    q.topicEmoji = topic.emoji;
    return q;
  },

  // ── TOPIC PROGRESS TRACKING ───────────────────────────────────────────
  // Returns per-topic accuracy from saved data
  getTopicStats() {
    try {
      return JSON.parse(localStorage.getItem('mb_topic_stats') || '{}');
    } catch(e) { return {}; }
  },
  recordResult(topicId, correct) {
    try {
      const stats = this.getTopicStats();
      if (!stats[topicId]) stats[topicId] = {correct:0, total:0};
      stats[topicId].total++;
      if (correct) stats[topicId].correct++;
      localStorage.setItem('mb_topic_stats', JSON.stringify(stats));
    } catch(e) {}
  },
  getWeakTopics(standard) {
    const stats = this.getTopicStats();
    const pool  = this.STANDARD_TOPICS[standard] || [];
    return pool.filter(t => {
      const s = stats[t];
      return s && s.total >= 3 && (s.correct / s.total) < 0.6;
    });
  },
};

    document.getElementById('optsGrid').style.display='grid';
    document.getElementById('npWrap').style.display='none';
    document.getElementById('abInline').className='ab-inline';
    document.getElementById('abInstruction').style.display='none';
    document.getElementById('abSubmitBtn').className='ab-submit';
    genOptions(ans);
  }
  // Auto-start timer immediately — no per-question overlay
  startTimer(calcTimer(a,b,op));

}
function genOptions(correct){
  const s=new Set([correct]);let att=0;
  while(s.size<4&&att<200){
    att++;let w;const r=Math.random();
    if(r<0.33){const off=randInt(1,Math.max(5,Math.floor(correct*0.08)));w=correct+(Math.random()<0.5?off:-off);}
    else if(r<0.66){const off=randInt(10,Math.max(20,Math.floor(correct*0.15)));w=correct+(Math.random()<0.5?off:-off);}
    else{w=correct+(Math.random()<0.5?1:-1)*randInt(Math.floor(correct*0.05)+1,Math.floor(correct*0.25)+10);}
    w=Math.round(w);if(w>0&&w!==correct&&!s.has(w))s.add(w);
  }
  const arr=shuffle([...s]);G.options=arr;
  for(let i=0;i<4;i++){const b=document.getElementById('o'+i);b.textContent=arr[i];b.className='opt-btn';b.disabled=false;}
}

function processResult(ok){
  G.totalAttempts++;const elapsed=G.timerMax-G.timerLeft;
  if(G.currentTopic)SUPA.logAttempt({topicId:G.currentTopic,topicLabel:G.currentTopicLabel,question:G.curA+' '+G.curOp+' '+G.curB,correct:ok,timeTaken:elapsed});
  if(ok){
    G.streak++;G.totalCorrect++;G.dailyCount++;
    if(G.streak>G.bestStreak)G.bestStreak=G.streak;
    if(elapsed<=5)G.fastAnswers++;if(G.mode==='abacus')G.abacusCorrect++;
    const sb=G.streak>=10?4:G.streak>=5?2:G.streak>=3?1:0;
    const xpG=12+sb*6+G.level*2,cG=3+sb+Math.floor(G.level*0.7);
    addXP(xpG);G.coins+=cG;showFeedback(true);
    showFloat('+'+xpG+' XP','#ffd700');
    if(sb>0)showFloat('🔥 ×'+(sb+1)+' BONUS!','#ff6b00');
    if(elapsed<=5)showFloat('⚡ SPEED!','#4fc3f7');
    if(G.mode==='boss'){G.bossHP--;updateBossBar();if(G.bossHP<=0){setTimeout(bossDead,800);return;}}
    checkAchievements();checkDaily();updateUI();saveProgress();
    setTimeout(()=>{clearFeedback();generateQuestion();},1100);
  }else{
    G.streak=0;G.lives--;showFeedback(false);showAnsReveal();shakeArena();
    updateLivesUI();checkAchievements();checkDaily();updateUI();
    if(G.lives<=0){setTimeout(gameOver,1800);return;}

  }
}

// ── ABACUS SUBMIT — reads current abacus total and checks against answer ──
function submitAbacus(){
  if(G.answering)return;
  // Read the current abacus total
  let total=0;
  AB_COLS.forEach((mult,ci)=>{
    let ec=abState.earth[ci].filter(v=>v).length;
    total+=(abState.heaven[ci]*5+ec)*mult;
  });
  G.answering=true;
  clearTimer();
  // Disable submit so can't tap again
  document.getElementById('abSubmitBtn').disabled=true;
  const ok=(total===G.curAnswer);
  // Flash the abacus inline border
  const ab=document.getElementById('abInline');
  ab.style.border='2px solid '+(ok?'var(--green)':'var(--red)');
  ab.style.boxShadow=ok?'0 0 16px rgba(0,230,118,0.4)':'0 0 16px rgba(255,68,68,0.4)';
  // If wrong, show what the correct abacus value should have been
  if(!ok){
    const hint=document.createElement('div');
    hint.style.cssText='text-align:center;font-size:0.85rem;font-weight:800;color:var(--red);margin-top:6px;';
    hint.textContent='Your abacus shows: '+total+' — Correct answer was: '+G.curAnswer;
    ab.appendChild(hint);
    setTimeout(()=>{if(ab.contains(hint))ab.removeChild(hint);},3500);
  }
  processResult(ok);
}
